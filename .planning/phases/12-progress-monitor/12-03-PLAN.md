```tsx
import React, { useState, useEffect } from 'react';

interface Phase {
  name: string;
  status: 'done' | 'active' | 'pending';
}

interface FileItem {
  name: string;
  path: string;
  canPreview: boolean;
  canOpen: boolean;
}

interface ProgressScreenProps {
  projectName: string;
  onPause?: () => void;
  onCancel?: () => void;
}

export default function ProgressScreen({ projectName, onPause, onCancel }: ProgressScreenProps) {
  const [phases, setPhases] = useState<Phase[]>([]);
  const [files, setFiles] = useState<FileItem[]>([]);
  const [statusMessage, setStatusMessage] = useState('Starting...');
  const [percentage, setPercentage] = useState(0);

  useEffect(() => {
    // Subscribe to IPC channels
    const unsubUpdate = window.electronAPI?.project?.onUpdate?.((data) => {
      setStatusMessage(data.message);
    });

    const unsubProgress = window.electronAPI?.project?.onProgress?.((data) => {
      setPhases(data.phases);
      // Calculate percentage
      const done = data.phases.filter(p => p.status === 'done').length;
      setPercentage(Math.round((done / data.phases.length) * 100));
    });

    const unsubFile = window.electronAPI?.project?.onFile?.((data) => {
      setFiles(prev => [...prev, data]);
    });

    return () => {
      unsubUpdate?.();
      unsubProgress?.();
      unsubFile?.();
    };
  }, []);

  const handleOpenFile = (path: string) => {
    window.electronAPI?.project?.openFile?.(path);
  };

  const handlePreviewFile = (path: string) => {
    // TODO: Open preview panel
    console.log('Preview:', path);
  };

  return (
    <div className="progress-screen" style={{ ... }}>
      <h1>Building: {projectName}</h1>

      {/* Progress bar */}
      <div className="progress-bar">
        <div className="progress-fill" style={{ width: `${percentage}%` }} />
      </div>
      <span>{percentage}%</span>

      {/* Phase tree */}
      <div className="phase-tree">
        {phases.map((phase, i) => (
          <div key={i} className={`phase phase-${phase.status}`}>
            <span className="phase-icon">
              {phase.status === 'done' ? '✓' : phase.status === 'active' ? '●' : '○'}
            </span>
            <span className="phase-name">{phase.name}</span>
          </div>
        ))}
      </div>

      {/* Status message */}
      <div className="status-message">{statusMessage}</div>

      {/* Files created */}
      {files.length > 0 && (
        <div className="files-list">
          <h3>Files created:</h3>
          {files.map((file, i) => (
            <div key={i} className="file-item">
              <span>{file.name}</span>
              <div className="file-actions">
                <button onClick={() => handleOpenFile(file.path)}>Open</button>
                {file.canPreview && (
                  <button onClick={() => handlePreviewFile(file.path)}>Preview</button>
                )}
              </div>
            </div>
          ))}
        </div>
      )}

      {/* Action buttons */}
      <div className="actions">
        {onPause && <button onClick={onPause}>Pause</button>}
        {onCancel && <button onClick={onCancel}>Cancel</button>}
      </div>
    </div>
  );
}
```

Styling:
- Progress bar: full width, 8px height, rounded
- Phase tree: vertical list with indentation
- Phase icons: ✓ (green), ● (blue), ○ (gray)
- Files list: horizontal layout per item (name left, buttons right)
- Match existing design language

```tsx
import React from 'react';

interface CompleteScreenProps {
  projectName: string;
  humanFolder: string;
  deployedUrl?: string;
  summary: {
    pages: number;
    components: number;
  };
  onNewProject: () => void;
  onModify: () => void;
}

export default function CompleteScreen({
  projectName,
  humanFolder,
  deployedUrl,
  summary,
  onNewProject,
  onModify,
}: CompleteScreenProps) {
  const handleOpenSite = () => {
    if (deployedUrl) {
      window.electronAPI?.project?.openUrl?.(deployedUrl);
    }
  };

  const handleOpenFinder = () => {
    window.electronAPI?.project?.openFolder?.(humanFolder);
  };

  const handleOpenVSCode = () => {
    // Open in VS Code via shell command
    window.electronAPI?.project?.openFile?.(humanFolder);
  };

  return (
    <div className="complete-screen" style={{ ... }}>
      <h1>✓ {projectName} — Built</h1>

      {deployedUrl && (
        <div className="deployed-section">
          <p>Live at: <a href="#" onClick={handleOpenSite}>{deployedUrl}</a></p>
          <button onClick={handleOpenSite}>Open Site</button>
        </div>
      )}

      <div className="folder-section">
        <p>Project folder:</p>
        <code>{humanFolder}</code>
        <div className="folder-actions">
          <button onClick={handleOpenFinder}>Open in Finder</button>
          <button onClick={handleOpenVSCode}>Open in VS Code</button>
        </div>
      </div>

      <div className="summary-section">
        <p>What was built:</p>
        <p>{summary.pages} pages, {summary.components} components</p>
      </div>

      <div className="actions">
        <button onClick={onNewProject} className="primary">New Project</button>
        <button onClick={onModify}>Modify This</button>
      </div>
    </div>
  );
}
```

- Deployed URL as clickable link
- Folder path in monospace
- Two-column action buttons at bottom
