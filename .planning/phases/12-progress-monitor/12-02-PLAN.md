   ```typescript
   import * as fs from 'fs';
   import * as path from 'path';
   import { BrowserWindow } from 'electron';
   import { translateStatus, translateSpine, friendlyFileName, generateReadMe } from './transposer';
   import { KENOKI_HIDDEN, KENOKI_VISIBLE } from './project-scaffold';
   ```

2. **FileBridge class:**
   ```typescript
   export class FileBridge {
     private watcher: FSWatcher | null = null;
     private projectId: string;
     private projectName: string;
     private agentRoot: string;
     private humanRoot: string;
     private mainWindow: BrowserWindow | null = null;

     constructor(projectId: string, projectName: string) { ... }

     setMainWindow(window: BrowserWindow) { ... }

     start(): void { ... }

     stop(): void { ... }
   }
   ```

3. **Constructor:**
   - Store projectId, projectName
   - Calculate agentRoot = `${KENOKI_HIDDEN}/${projectId}`
   - Calculate humanRoot = `${KENOKI_VISIBLE}/${projectName}`

4. **start() method:**
   - Create watcher with chokidar:
     ```typescript
     this.watcher = chokidar.watch(this.agentRoot, {
       persistent: true,
       ignoreInitial: false,
       awaitWriteFinish: { stabilityThreshold: 100, pollInterval: 50 },
       ignored: /(^|[\/\\])\../,  // Ignore dotfiles
     });
     ```
   - Register handlers:
     - `this.watcher.on('add', (filePath) => this.handleFileCreated(filePath))`
     - `this.watcher.on('change', (filePath) => this.handleFileChanged(filePath))`

5. **handleFileCreated(agentPath: string):**
   - If path includes `/workspace/`:
     - Map to human path using `mapToHumanPath()`
     - Create parent directory with `fs.mkdirSync(dir, { recursive: true })`
     - Copy file with `fs.copyFileSync(agentPath, humanPath)`
     - Emit IPC `project:file`:
       ```typescript
       this.notify({
         type: 'project:file',
         data: {
           name: friendlyFileName(agentPath),
           path: humanPath,
           canPreview: this.canPreview(agentPath),
           canOpen: true,
         },
       });
       ```

6. **handleFileChanged(agentPath: string):**
   - If path includes `/status/` and ends with `.json`:
     - Read file, parse JSON
     - Call `translateStatus(status)`
     - Emit IPC `project:update`
   - If path includes `spine_master.md` or `spine_record.md`:
     - Read file content
     - Call `translateSpine(content)`
     - Emit IPC `project:progress`
     - Check if project complete with `isProjectComplete()`
   - If path includes `/workspace/`:
     - Map and copy to human folder (sync)
   - If path includes `/skills/active_mcps.json`:
     - Check for `missing` array
     - If non-empty, emit IPC `project:action` with MCP connection prompt

7. **Helper methods:**
   - `mapToHumanPath(agentPath: string): string` — Maps workspace path to Files/ path
   - `canPreview(path: string): boolean` — Returns true for images (.png, .jpg, .gif, .svg) and markdown
   - `isProjectComplete(spineContent: string): boolean` — All phases have [COMPLETE]
   - `handleProjectComplete(): void` — Final sync, generate Read Me, emit `project:complete`
   - `reconcileAllFiles(): void` — Sync all workspace files to human folder

8. **notify(update: { type: string; data: unknown }):**
   - Guard: `if (!this.mainWindow) return;`
   - `this.mainWindow.webContents.send(update.type, update.data);`

9. **stop() method:**
   - Close watcher: `this.watcher?.close()`
   - Set watcher to null

chokidar is already a dependency (used in src/main/file-watcher.ts).
